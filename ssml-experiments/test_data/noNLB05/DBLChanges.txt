# Biblica DBL Changes 2.0
#===========================================================================================
#
# The following changes simplify the characters and markup used in a DBL bundle
#
#===========================================================================================
# Change modifier apostrophe to close single quote in case the font does not support u02BC 
'\u02BC' > '’'
# Change horizontal line to to emdash in case the font does not support u2015 
#'\u2015' > '—'
# Change tilde to no break space
'~' > '\u00A0'
# Remove optional break markup
'//' > ''
# Remove \fm around footnotes
'\\fm\s+(\\f .*?\\f\*)\\fm\*' > '\1'
# Remove \fm dedup\fm*
'\\fm\s+(?i)dedup\\fm\*' > '\1'
# Remove comma between footnote callers \fm ,\fm*
#'\\fm\s,\fm\*' > ''
#
# Use \vp Publishing verses  
#'(?<=\\v\s)\S+(\s*)\\vp\s+([^\\]*?)\\vp\*\s*' > '\2\1'
# Remove \vp Publishing verses if they exist 
'\\vp\s+[^\\]*?\\vp\*\s*' > ''
#
# Use \fl for footnote references 
#'(?<=\\fr\s)[^\\]*?\\fl\s+' > ''
# Remove \fl after footnote references
'(?<=\\fr\s[^\\]*?)\\fl\s+[^\\]*?(?=\\)' > ''
#
# Remove TBOTB Book introduction
#'(?i)(?s)\s\\rem TBOTB Book Introduction.*?(?<=\s?\\ie\s)' > ''
#
# Remove CHAPTER AND VERSE NUMBERS AND OPTIONAL \fl ON CHAPTER LEVEL FOOTNOTES
#'(?<=\\fr\s+)\d+:0\s*(\\fl\s)?' > ''
#
# DISPLAY A GLOSSARY WORD WITH AN PRECEEDING ASTERIX
#'\\w\s(.*?)(\|.*?)?\\w\*'    >   '\1*'
# Remove Wordlist markup
'\\w\s([^\\|].*?)(\|[^\\].*?)?\\w\*' > '\1'
#
# ADD A THIN SPACE EITHER SIDE OF ELIPSES, EM-DASH AND AND FIGURE DASH
# Verify that the font supports thin spaces
#'(?<=\S)[\u2014\u2026\u2012](?=\S)' > '\u2006\0\u2006'
#'(?<=\s)[\u2014\u2026\u2012](?=\S)' > '\0\u2006'
#'(?<=\S)[\u2014\u2026\u2012](?=\s)' > '\u2006\0'
#change thin spaces either side of dash or elipsis to regular spaces
'(?<=[\u2012-\u2014\u2026])[\u2006\u2009]|[\u2006\u2009](?=[\u2012-\u2014\u2026])' > ' '
# Change \nd to \sc
#'(?<=\\\+?)nd(?=[\s\*])' > 'sc'
# Remove CHAPTER NUMBER IN FOOTNOTES
#'(?<=\\fr\s+)\d+:' > ''
# Remove VERSE 1 unless preceded by \cl
#'(?<=\\c \d+\s+(\\(\+)?(?!(cl))[^\\]*)+)\\v 1([-,]\d+)? ' > ''
# Remove Footnotes and parallel passage references (for audio scripts)
#'\\f .*?\\f\*' > ''
#'\\r .*?[\r\n]+' > ''
# Remove all book introductions (for audio scripts)
#'(?<=\\mt\w*\s[^\\]*?(?=\\i))(?s).*?(?=\\c\s)' > ''
# Remove Standard Book introduction
#'(?i)(?s)\s\\i\w.*?\\rem TBOTB Book Introduction' > ''
# Remove \imt1 (use \imt only for electronic editions)
#'\s+\\imt1 .*?(?=\s\\)' > ''
